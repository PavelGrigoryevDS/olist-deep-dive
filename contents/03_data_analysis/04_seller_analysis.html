def seller_top(metric: str, show_cnt: bool=True, ascending=False):
    """Show Top Customers by Metric"""
    cols = ['seller_id', metric]
    if show_cnt:
        if metric == 'products_cnt':
            cols += ['orders_cnt']
        else:
            cols += ['products_cnt', 'orders_cnt']
    display(
        df_sellers[cols]
        .sort_values(metric, ascending=ascending)
        .set_index('seller_id')
        .head(10)
    )
